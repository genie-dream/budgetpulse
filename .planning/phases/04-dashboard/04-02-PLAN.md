const paceStatus = getPaceStatus(paceRatio)

// Number color: red only when danger
const heroAmountColor =
  paceStatus === 'danger' || remainingBudget < 0 ? 'text-red-500' : 'text-slate-100'

// Progress bar color
const progressBarColor =
  paceStatus === 'safe' ? 'bg-green-500' :
  paceStatus === 'caution' ? 'bg-amber-400' :
  'bg-red-500'

// Badge color
const badgeBgColor =
  paceStatus === 'safe' ? 'bg-green-500/20' :
  paceStatus === 'caution' ? 'bg-amber-400/20' :
  'bg-red-500/20'
const badgeTextColor =
  paceStatus === 'safe' ? 'text-green-400' :
  paceStatus === 'caution' ? 'text-amber-400' :
  'text-red-500'

// Progress percent: clamped 0–100, always 100 when over-budget
const progressPct = remainingBudget < 0
  ? 100
  : Math.min(100, Math.max(0, variableBudget > 0 ? (totalSpent / variableBudget) * 100 : 0))

// Label: "Over budget" when negative, else "remaining this month"
const heroLabel = remainingBudget < 0 ? t('overBudget') : t('remainingThisMonth')
